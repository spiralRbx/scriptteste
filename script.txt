-- Scripts Locais
local function createScript(name, button, func)
    local script = Instance.new("LocalScript")
    script.Name = name
    script.Parent = button
    script.Source = func
end

createScript("LocalScript1", Converted["bypass1"], [[
    local Players = game:GetService("Players")
    local StarterPack = game:GetService("StarterPack")
    local player = Players.LocalPlayer
    local boomBox = StarterPack:WaitForChild("BoomBox")

    local function adicionarBoomBoxAoInventario()
        local novaBoomBox = boomBox:Clone()
        if novaBoomBox:FindFirstChildOfClass("Script") or novaBoomBox:FindFirstChildOfClass("LocalScript") then
            novaBoomBox.Name = "BoomBoxDuplicado"
            novaBoomBox.Parent = player:WaitForChild("Backpack")
        else
            warn("A BoomBox duplicada não contém scripts necessários.")
        end
    end

    adicionarBoomBoxAoInventario()
]])

createScript("LocalScript2", Converted["bypass2"], [[
    local Players = game:GetService("Players")
    local player = Players.LocalPlayer
    local backpack = player:WaitForChild("Backpack")

    local function equiparTodasAsBoomBox()
        for _, item in pairs(backpack:GetChildren()) do
            if item:IsA("Tool") and item.Name:find("BoomBox") then
                player.Character.Humanoid:EquipTool(item)
            end
        end
    end

    equiparTodasAsBoomBox()
]])

createScript("LocalScript3", Converted["bypass3"], [[
    local Players = game:GetService("Players")
    local player = Players.LocalPlayer
    local backpack = player:WaitForChild("Backpack")
    local chatService = require(game:GetService("Chat"))

    local function girarBoomBox()
        local distancia = 10
        local velocidade = 5

        for _, item in pairs(backpack:GetChildren()) do
            if item:IsA("Tool") and item.Name:find("BoomBox") then
                local posicaoInicial = item.Position
                local tempo = 0

                local rotacaoCentro = Instance.new("Part")
                rotacaoCentro.Size = Vector3.new(1, 1, 1)
                rotacaoCentro.Position = player.Character.HumanoidRootPart.Position
                rotacaoCentro.Anchored = true
                rotacaoCentro.CanCollide = false
                rotacaoCentro.Parent = workspace

                while tempo < 2 * math.pi do
                    tempo = tempo + (math.pi / velocidade) * game:GetService("RunService").RenderStepped:Wait()
                    local x = distancia * math.cos(tempo)
                    local z = distancia * math.sin(tempo)
                    item.Position = rotacaoCentro.Position + Vector3.new(x, 0, z)
                end

                rotacaoCentro:Destroy()
            end
        end
    end

    local function onChatMessage(message)
        if message:lower() == "!animarboombox" then
            girarBoomBox()
        end
    end

    player.Chatted:Connect(onChatMessage)
]])
